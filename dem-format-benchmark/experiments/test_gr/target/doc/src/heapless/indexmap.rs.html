15/// A [`heaples::IndexMap`](./struct.IndexMap.html) using the default FNV hasher
16///
17/// A list of all Methods and Traits available for `FnvIndexMap` can be found in
18/// the [`heapless::IndexMap`](./struct.IndexMap.html) documentation.
19///
20/// # Examples
21/// ```
22/// use heapless::FnvIndexMap;
23/// use heapless::consts::*;
24///
25/// // A hash map with a capacity of 16 key-value pairs allocated on the stack
26/// let mut book_reviews = FnvIndexMap::<_, _, U16>::new();
27///
28/// // review some books.
29/// book_reviews.insert("Adventures of Huckleberry Finn",    "My favorite book.").unwrap();
30/// book_reviews.insert("Grimms' Fairy Tales",               "Masterpiece.").unwrap();
31/// book_reviews.insert("Pride and Prejudice",               "Very enjoyable.").unwrap();
32/// book_reviews.insert("The Adventures of Sherlock Holmes", "Eye lyked it alot.").unwrap();
33///
34/// // check for a specific one.
35/// if !book_reviews.contains_key("Les Misérables") {
36///     println!("We've got {} reviews, but Les Misérables ain't one.",
37///              book_reviews.len());
38/// }
39///
40/// // oops, this review has a lot of spelling mistakes, let's delete it.
41/// book_reviews.remove("The Adventures of Sherlock Holmes");
42///
43/// // look up the values associated with some keys.
44/// let to_find = ["Pride and Prejudice", "Alice's Adventure in Wonderland"];
45/// for book in &to_find {
46///     match book_reviews.get(book) {
47///         Some(review) => println!("{}: {}", book, review),
48///         None => println!("{} is unreviewed.", book)
49///     }
50/// }
51///
52/// // iterate over everything.
53/// for (book, review) in &book_reviews {
54///     println!("{}: \"{}\"", book, review);
55/// }
56/// ```
57pub type FnvIndexMap<K, V, N> = IndexMap<K, V, N, BuildHasherDefault<FnvHasher>>;

59#[derive(Clone, Copy, Eq, PartialEq)]
60struct HashValue(u16);

62impl HashValue {
63    fn desired_pos(&self, mask: usize) -> usize {
64        usize::from(self.0) & mask
65    }
66
67    fn probe_distance(&self, mask: usize, current: usize) -> usize {
68        current.wrapping_sub(self.desired_pos(mask) as usize) & mask
69    }
70}

81#[derive(Clone, Copy, PartialEq)]
82pub struct Pos {
83    // compact representation of `{ hash_value: u16, index: u16 }`
84    // To get the most from `NonZero` we store the *value minus 1*. This way `None::Option<Pos>`
85    // is equivalent to the very unlikely value of  `{ hash_value: 0xffff, index: 0xffff }` instead
86    // the more likely of `{ hash_value: 0x00, index: 0x00 }`
87    nz: NonZeroU32,
88}
89
90impl Pos {
91    fn new(index: usize, hash: HashValue) -> Self {
92        Pos {
93            nz: unsafe {
94                NonZeroU32::new_unchecked(
95                    ((u32::from(hash.0) << 16) + index as u32).wrapping_add(1),
96                )
97            },
98        }
99    }
100
101    fn hash(&self) -> HashValue {
102        HashValue((self.nz.get().wrapping_sub(1) >> 16) as u16)
103    }
104
105    fn index(&self) -> usize {
106        self.nz.get().wrapping_sub(1) as u16 as usize
107    }
108}

110pub enum Inserted<V> {
111    Done,
112    Swapped { prev_value: V },
113    RobinHood { probe: usize, old_pos: Pos },
114}
115
116macro_rules! probe_loop {
117    ($probe_var: ident < $len: expr, $body: expr) => {
118        loop {
119            if $probe_var < $len {
120                $body
121                    $probe_var += 1;
122            } else {
123                $probe_var = 0;
124            }
125        }
126    }
127}

129struct CoreMap<K, V, N>
130where
131    K: Eq + Hash,
132    N: ArrayLength<Bucket<K, V>> + ArrayLength<Option<Pos>>,
133{
134    entries: Vec<Bucket<K, V>, N>,
135    indices: GenericArray<Option<Pos>, N>,
136}
137
138impl<K, V, N> CoreMap<K, V, N>
139where
140    K: Eq + Hash,
141    N: ArrayLength<Bucket<K, V>> + ArrayLength<Option<Pos>>,
142{
143    // TODO turn into a `const fn`; needs `mem::zeroed` to be a `const fn`
144    fn new() -> Self {
145        CoreMap {
146            entries: Vec::new(),
147            indices: unsafe { MaybeUninit::zeroed().assume_init() },
148        }
149    }
150
151    fn capacity() -> usize {
152        N::to_usize()
153    }
154
155    fn mask() -> usize {
156        Self::capacity() - 1
157    }
158
159    fn find<Q>(&self, hash: HashValue, query: &Q) -> Option<(usize, usize)>
160    where
161        K: Borrow<Q>,
162        Q: ?Sized + Eq,
163    {
164        let mut probe = hash.desired_pos(Self::mask());
165        let mut dist = 0;
166
167        probe_loop!(probe < self.indices.len(), {
168            if let Some(pos) = self.indices[probe] {
169                let entry_hash = pos.hash();
170                // NOTE(i) we use unchecked indexing below
171                let i = pos.index();
172                debug_assert!(i < self.entries.len());
173
174                if dist > entry_hash.probe_distance(Self::mask(), probe) {
175                    // give up when probe distance is too long
176                    return None;
177                } else if entry_hash == hash
178                    && unsafe { self.entries.get_unchecked(i).key.borrow() == query }
179                {
180                    return Some((probe, i));
181                }
182            } else {
183                return None;
184            }
185
186            dist += 1;
187        });
188    }
189
190    // First phase: Look for the preferred location for key.
191    //
192    // We will know if `key` is already in the map, before we need to insert it.
193    // When we insert they key, it might be that we need to continue displacing
194    // entries (robin hood hashing), in which case Inserted::RobinHood is returned
195    fn insert_phase_1(&mut self, hash: HashValue, key: K, value: V) -> Inserted<V> {
196        let mut probe = hash.desired_pos(Self::mask());
197        let mut dist = 0;
198
199        let inserted;
200        probe_loop!(probe < self.indices.len(), {
201            let pos = &mut self.indices[probe];
202
203            if let Some(pos) = *pos {
204                let entry_hash = pos.hash();
205                // NOTE(i) we use unchecked indexing below
206                let i = pos.index();
207                debug_assert!(i < self.entries.len());
208
209                let their_dist = entry_hash.probe_distance(Self::mask(), probe);
210
211                if their_dist < dist {
212                    // robin hood: steal the spot if it's better for us
213                    let index = self.entries.len();
214                    inserted = Inserted::RobinHood {
215                        probe: probe,
216                        old_pos: Pos::new(index, hash),
217                    };
218                    break;
219                } else if entry_hash == hash && unsafe { self.entries.get_unchecked(i).key == key }
220                {
221                    return Inserted::Swapped {
222                        prev_value: mem::replace(
223                            unsafe { &mut self.entries.get_unchecked_mut(i).value },
224                            value,
225                        ),
226                    };
227                }
228            } else {
229                // empty bucket, insert here
230                let index = self.entries.len();
231                *pos = Some(Pos::new(index, hash));
232                inserted = Inserted::Done;
233                break;
234            }
235            dist += 1;
236        });
237
238        // NOTE(unsafe) we already checked (in `insert`) that we aren't exceeding the capacity
239        unsafe { self.entries.push_unchecked(Bucket { hash, key, value }) }
240        inserted
241    }
242
243    // phase 2 is post-insert where we forward-shift `Pos` in the indices.
244    fn insert_phase_2(&mut self, mut probe: usize, mut old_pos: Pos) {
245        probe_loop!(probe < self.indices.len(), {
246            let pos = unsafe { self.indices.get_unchecked_mut(probe) };
247
248            let mut is_none = true; // work around lack of NLL
249            if let Some(pos) = pos.as_mut() {
250                old_pos = mem::replace(pos, old_pos);
251                is_none = false;
252            }
253
254            if is_none {
255                *pos = Some(old_pos);
256                break;
257            }
258        });
259    }
260
261    fn remove_found(&mut self, probe: usize, found: usize) -> (K, V) {
262        // index `probe` and entry `found` is to be removed
263        // use swap_remove, but then we need to update the index that points
264        // to the other entry that has to move
265        self.indices[probe] = None;
266        let entry = unsafe { self.entries.swap_remove_unchecked(found) };
267
268        // correct index that points to the entry that had to swap places
269        if let Some(entry) = self.entries.get(found) {
270            // was not last element
271            // examine new element in `found` and find it in indices
272            let mut probe = entry.hash.desired_pos(Self::mask());
273
274            probe_loop!(probe < self.indices.len(), {
275                if let Some(pos) = self.indices[probe] {
276                    if pos.index() >= self.entries.len() {
277                        // found it
278                        self.indices[probe] = Some(Pos::new(found, entry.hash));
279                        break;
280                    }
281                }
282            });
283        }
284
285        self.backward_shift_after_removal(probe);
286
287        (entry.key, entry.value)
288    }
289
290    fn backward_shift_after_removal(&mut self, probe_at_remove: usize) {
291        // backward shift deletion in self.indices
292        // after probe, shift all non-ideally placed indices backward
293        let mut last_probe = probe_at_remove;
294        let mut probe = probe_at_remove + 1;
295
296        probe_loop!(probe < self.indices.len(), {
297            if let Some(pos) = self.indices[probe] {
298                let entry_hash = pos.hash();
299
300                if entry_hash.probe_distance(Self::mask(), probe) > 0 {
301                    unsafe { *self.indices.get_unchecked_mut(last_probe) = self.indices[probe] }
302                    self.indices[probe] = None;
303                } else {
304                    break;
305                }
306            } else {
307                break;
308            }
309            last_probe = probe;
310        });
311    }
312}

328/// Fixed capacity [`IndexMap`](https://docs.rs/indexmap/1/indexmap/map/struct.IndexMap.html)
329///
330/// Note that you cannot use `IndexMap` directly, since it is generic around the hashing algorithm
331/// in use. Pick a concrete instantiation like [`FnvIndexMap`](./type.FnvIndexMap.html) instead
332/// or create your own.
333///
334/// Note that the capacity of the `IndexMap` must be a power of 2.

338/// for this example.

353/// // check for a specific one.
354/// if !book_reviews.contains_key("Les Misérables") {
355///     println!("We've got {} reviews, but Les Misérables ain't one.",
356///              book_reviews.len());
357/// }
358///

630    /// Returns a mutable reference to the value corresponding to the key.
631    ///
632    /// The key may be any borrowed form of the map's key type, but `Hash` and `Eq` on the borrowed
633    /// form *must* match those for the key type.
634    ///
635    /// Computes in **O(1)** time (average).
636    ///
637    /// # Examples
638    ///
639    /// ```
640    /// use heapless::FnvIndexMap;
641    /// use heapless::consts::*;
642    ///
643    /// let mut map = FnvIndexMap::<_, _, U8>::new();

645    /// if let Some(x) = map.get_mut(&1) {

651    where

654    {
655        if let Some((_, found)) = self.find(key) {
656            Some(unsafe { &mut self.core.entries.get_unchecked_mut(found).value })
657        } else {
658            None

662    /// Inserts a key-value pair into the map.
663    ///
664    /// If an equivalent key already exists in the map: the key remains and retains in its place in
665    /// the order, its corresponding value is updated with `value` and the older value is returned
666    /// inside `Some(_)`.
667    ///
668    /// If no equivalent key existed in the map: the new key-value pair is inserted, last in order,
669    /// and `None` is returned.
670    ///
671    /// Computes in **O(1)** time (average).
672    ///
673    /// See also entry if you you want to insert or modify or if you need to get the index of the
674    /// corresponding key-value pair.
675    ///
676    /// # Examples
677    ///
678    /// ```
679    /// use heapless::FnvIndexMap;
680    /// use heapless::consts::*;
681    ///
682    /// let mut map = FnvIndexMap::<_, _, U8>::new();
683    /// assert_eq!(map.insert(37, "a"), Ok(None));
684    /// assert_eq!(map.is_empty(), false);
685    ///
686    /// map.insert(37, "b");
687    /// assert_eq!(map.insert(37, "c"), Ok(Some("b")));
688    /// assert_eq!(map[&37], "c");
689    /// ```
690    pub fn insert(&mut self, key: K, value: V) -> Result<Option<V>, (K, V)> {
691        if self.core.entries.is_full() {
692            Err((key, value))
693        } else {
694            Ok(match self.insert_phase_1(key, value) {
695                Inserted::Swapped { prev_value } => Some(prev_value),
696                Inserted::Done => None,
697                Inserted::RobinHood { probe, old_pos } => {
698                    self.core.insert_phase_2(probe, old_pos);
699                    None
700                }
701            })
702        }
703    }
704
705    /// Same as [`swap_remove`](struct.IndexMap.html#method.swap_remove)
706    ///
707    /// Computes in **O(1)** time (average).
708    ///
709    /// # Examples
710    ///
711    /// ```
712    /// use heapless::FnvIndexMap;
713    /// use heapless::consts::*;
714    ///
715    /// let mut map = FnvIndexMap::<_, _, U8>::new();
716    /// map.insert(1, "a").unwrap();
717    /// assert_eq!(map.remove(&1), Some("a"));
718    /// assert_eq!(map.remove(&1), None);
719    /// ```
720    pub fn remove<Q>(&mut self, key: &Q) -> Option<V>
721    where
722        K: Borrow<Q>,
723        Q: ?Sized + Hash + Eq,
724    {
725        self.swap_remove(key)
726    }
727
728    /// Remove the key-value pair equivalent to `key` and return its value.
729    ///
730    /// Like `Vec::swap_remove`, the pair is removed by swapping it with the last element of the map
731    /// and popping it off. **This perturbs the postion of what used to be the last element!**
732    ///
733    /// Return `None` if `key` is not in map.
734    ///
735    /// Computes in **O(1)** time (average).
736    pub fn swap_remove<Q>(&mut self, key: &Q) -> Option<V>
737    where
738        K: Borrow<Q>,
739        Q: ?Sized + Hash + Eq,
740    {
741        self.find(key)
742            .map(|(probe, found)| self.core.remove_found(probe, found).1)
743    }
744
745    /* Private API */
746    /// Return probe (indices) and position (entries)
747    fn find<Q>(&self, key: &Q) -> Option<(usize, usize)>
748    where
749        K: Borrow<Q>,
750        Q: ?Sized + Hash + Eq,
751    {
752        if self.len() == 0 {
753            return None;
754        }
755        let h = hash_with(key, &self.build_hasher);
756        self.core.find(h, key)
757    }
758
759    fn insert_phase_1(&mut self, key: K, value: V) -> Inserted<V> {
760        let hash = hash_with(&key, &self.build_hasher);
761        self.core.insert_phase_1(hash, key, value)
762    }
763}

967fn hash_with<K, S>(key: &K, build_hasher: &S) -> HashValue
968where
969    K: ?Sized + Hash,
970    S: BuildHasher,
