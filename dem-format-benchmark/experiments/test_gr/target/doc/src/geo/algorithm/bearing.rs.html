3/// Returns the bearing to another Point in degrees.
4///
5/// Bullock, R.: Great Circle Distances and Bearings Between Two Locations, 2007.
6/// (https://dtcenter.org/met/users/docs/write_ups/gc_simple.pdf)
7
8pub trait Bearing<T: CoordFloat> {
9    /// Returns the bearing to another Point in degrees, where North is 0° and East is 90°.
10    ///
11    /// # Examples
12    ///
13    /// ```
14    /// # #[macro_use] extern crate approx;
15    /// #
16    /// use geo::algorithm::bearing::Bearing;
17    /// use geo::Point;
18    ///
19    /// let p_1 = Point::<f64>::new(9.177789688110352, 48.776781529534965);
20    /// let p_2 = Point::<f64>::new(9.274410083250379, 48.84033282787534);
21    /// let bearing = p_1.bearing(p_2);
22    /// assert_relative_eq!(bearing, 45., epsilon = 1.0e-6);
23    /// ```
24    fn bearing(&self, point: Point<T>) -> T;
25}
26
27impl<T> Bearing<T> for Point<T>
28where
29    T: CoordFloat,
30{
31    fn bearing(&self, point: Point<T>) -> T {
32        let (lng_a, lat_a) = (self.x().to_radians(), self.y().to_radians());
33        let (lng_b, lat_b) = (point.x().to_radians(), point.y().to_radians());
34        let delta_lng = lng_b - lng_a;
35        let s = lat_b.cos() * delta_lng.sin();
36        let c = lat_a.cos() * lat_b.sin() - lat_a.sin() * lat_b.cos() * delta_lng.cos();
37
38        T::atan2(s, c).to_degrees()
39    }
40}

42#[cfg(test)]
43mod test {
44    use crate::algorithm::bearing::Bearing;
45    use crate::algorithm::haversine_destination::HaversineDestination;
46    use crate::point;
47
48    #[test]
49    fn returns_the_proper_bearing_to_another_point() {
50        let p_1 = point!(x: 9.177789688110352f64, y: 48.776781529534965);
51        let p_2 = p_1.haversine_destination(45., 10000.);
52        let b_1 = p_1.bearing(p_2);
53        assert_relative_eq!(b_1, 45., epsilon = 1.0e-6);
54
55        let p_3 = point!(x: 9., y: 47.);
56        let p_4 = point!(x: 9., y: 48.);
57        let b_2 = p_3.bearing(p_4);
58        assert_relative_eq!(b_2, 0., epsilon = 1.0e-6);
59    }
