11    T: GeoNum,

13    fn contains(&self, coord: &Coordinate<T>) -> bool {
14        if self.start == self.end {
15            &self.start == coord
16        } else {

31impl<T> Contains<Line<T>> for Line<T>
32where
33    T: GeoNum,
34{
35    fn contains(&self, line: &Line<T>) -> bool {
36        if line.start == line.end {
37            self.contains(&line.start)
38        } else {
39            self.intersects(&line.start) && self.intersects(&line.end)
40        }
41    }
42}

44impl<T> Contains<LineString<T>> for Line<T>
45where
46    T: GeoNum,
47{
48    fn contains(&self, linestring: &LineString<T>) -> bool {
49        // Empty linestring has no interior, and not
50        // contained in anything.
51        if linestring.0.is_empty() {
52            return false;
53        }
54
55        // The interior of the linestring should have some
56        // intersection with the interior of self. Two cases
57        // arise:
58        //
59        // 1. There are at least two distinct points in the
60        // linestring. Then, if both intersect, the interior
61        // between these two must have non-empty intersection.
62        //
63        // 2. Otherwise, all the points on the linestring
64        // are the same. In this case, the interior is this
65        // specific point, and it should be contained in the
66        // line.
67        let first = linestring.0.first().unwrap();
68        let mut all_equal = true;
69
70        // If all the vertices of the linestring intersect
71        // self, then the interior or boundary of the
72        // linestring cannot have non-empty intersection
73        // with the exterior.
74        let all_intersects = linestring.0.iter().all(|c| {
75            if c != first {
76                all_equal = false;
77            }
78            self.intersects(c)
79        });
80
81        all_intersects && (!all_equal || self.contains(first))
82    }
83}
