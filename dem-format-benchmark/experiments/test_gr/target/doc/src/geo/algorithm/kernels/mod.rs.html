1use crate::{CoordNum, Coordinate};
2use num_traits::Zero;
3
4#[derive(Debug, PartialEq, Eq, PartialOrd, Ord, Clone, Copy)]
5pub enum Orientation {
6    CounterClockwise,
7    Clockwise,
8    Collinear,
9}
10
11/// Kernel trait to provide predicates to operate on
12/// different scalar types.
13pub trait Kernel<T: CoordNum> {
14    /// Gives the orientation of 3 2-dimensional points:
15    /// ccw, cw or collinear (None)
16    fn orient2d(p: Coordinate<T>, q: Coordinate<T>, r: Coordinate<T>) -> Orientation {
17        let res = (q.x - p.x) * (r.y - q.y) - (q.y - p.y) * (r.x - q.x);
18        if res > Zero::zero() {
19            Orientation::CounterClockwise
20        } else if res < Zero::zero() {
21            Orientation::Clockwise
22        } else {
23            Orientation::Collinear
24        }
25    }
26
27    fn square_euclidean_distance(p: Coordinate<T>, q: Coordinate<T>) -> T {
28        (p.x - q.x) * (p.x - q.x) + (p.y - q.y) * (p.y - q.y)
29    }
30
31    /// Compute the sign of the dot product of `u` and `v` using
32    /// robust predicates. The output is `CounterClockwise` if
33    /// the sign is positive, `Clockwise` if negative, and
34    /// `Collinear` if zero.
35    fn dot_product_sign(u: Coordinate<T>, v: Coordinate<T>) -> Orientation {
36        let zero = Coordinate::zero();
37        let vdash = Coordinate {
38            x: T::zero() - v.y,
39            y: v.x,
40        };
41        Self::orient2d(zero, u, vdash)
42    }
43}
44
45/// Marker trait to assign Kernel for scalars
46pub trait HasKernel: CoordNum {
47    type Ker: Kernel<Self>;
48}
49
50// Helper macro to implement `HasKernel` on a a scalar type
51// `T` (first arg.) by assigning the second arg. It expects
52// the second arg. to be a type that takes one generic
53// parameter that is `T`.
54#[macro_use]
55macro_rules! has_kernel {
56    ($t:ident, $k:ident) => {
57        impl $crate::algorithm::kernels::HasKernel for $t {
58            type Ker = $k;
59        }
60    };
61}
