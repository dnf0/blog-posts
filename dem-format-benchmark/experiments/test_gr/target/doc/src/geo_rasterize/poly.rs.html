8fn y_coordinates<'a>(
9    first: &'a LineString<f64>,
10    rest: &'a [LineString<f64>],
11) -> impl Iterator<Item = isize> + 'a {
12    once(first).chain(rest).flat_map(|line_string| {
13        line_string
14            .points_iter()
15            .map(|point| point.y().floor() as isize)
16    })
17}

19type PointPair = (Point<f64>, Point<f64>);

21/// Iterate over all points in a polygon (both the exterior and all
22/// the holes) in clockwise order and copy all succeding pairs of
23/// points into a Vec, like GDAL does.
24fn into_pointpairs(first: &LineString<f64>, rest: &[LineString<f64>]) -> Vec<PointPair> {
25    // I thought I could avoid allocating into a separate copy like
26    // GDAL does by doing some clever iterator tricks, but it turns
27    // out that we need to scan this sequence so many times that the
28    // perf loss from using iterators was worth 2x performance.
29    let num_pairs = once(first).chain(rest).map(|ls| ls.0.len() - 1).sum();
30    let mut result = Vec::with_capacity(num_pairs);
31
32    for ls in once(first).chain(rest) {
33        if ls.is_cw() {
34            result.extend(ls.points_iter().tuple_windows::<PointPair>());
35        } else {
36            result.extend(ls.points_iter().rev().tuple_windows::<PointPair>());
37        }
38    }
39
40    result
41}

53/// This will rasterize the interior of a polygon. We treat the
54/// arguments as LinearRings, which don't exist in `geo`, so we use
55/// closed `LineString`s instead.
56pub fn rasterize_polygon<Label>(
57    first: &LineString<f64>,
58    rest: &[LineString<f64>],
59    rasterizer: &mut Rasterizer<Label>,
60) where
61    Label: Copy + Add<Output = Label>,
62{
63    assert!(first.is_closed() && rest.iter().all(|ls| ls.is_closed()));
64    let total_points = first.coords_count()
65        + rest
66            .iter()
67            .map(|line_string| line_string.coords_count())
68            .sum::<usize>();
69    if total_points == 0 {
70        return;
71    }
72
73    // We'll reuse this Vec for each scanline; it will never more than
74    // `total_points` elements.
75    let mut xs: Vec<isize> = Vec::with_capacity(total_points);
76
77    // GDAL ensures that closed linear rings are always in CW order in
78    // `GDALCollectRingsFromGeometry`.
79    // These unwraps are safe since we would've returned if there were
80    // no points, so there must be at least one point.
81    let min_y = y_coordinates(first, rest).min().unwrap().max(0);
82    let max_y = y_coordinates(first, rest)
83        .max()
84        .unwrap()
85        .min(rasterizer.height() as isize - 1);
86    let min_x = 0;
87    let max_x = rasterizer.width() - 1;
88
89    let cw_points = into_pointpairs(first, rest);
90
91    // iterate over scanlines
92    for y in min_y..=max_y {
93        let dy = 0.5 + (y as f64); // center height of line
94
95        // iterate over ring point pairs
96        for (ind1, ind2) in cw_points.iter() {
97            // FIXME: we're assuming here that all loops are closed so
98            // that the last point == the first point; the gdal code
99            // makes this true.
100
101            // ind1,ind2 are a pair of adjacent points looping back to
102            // the front in the last case: (p0,p1), (p1,p2), (p2,p0)
103            let mut dy1 = ind1.y();
104            let mut dy2 = ind2.y();
105
106            if (dy1 < dy && dy2 < dy) || (dy1 > dy && dy2 > dy) {
107                continue;
108            }
109
110            let (dx1, dx2) = if dy1 < dy2 {
111                (ind1.x(), ind2.x())
112            } else if dy1 > dy2 {
113                std::mem::swap(&mut dy1, &mut dy2);
114                (ind2.x(), ind1.x())
115            } else {
116                if ind1.x() > ind2.x() {
117                    let horizontal_x1 = (ind2.x() + 0.5).floor() as isize;
118                    let horizontal_x2 = (ind1.x() + 0.5).floor() as isize;
119                    if horizontal_x1 > (max_x as isize) || horizontal_x2 <= min_x {
120                        continue;
121                    }
122                    rasterizer.fill_horizontal_line(
123                        horizontal_x1 as usize,
124                        horizontal_x2 as usize,
125                        y as usize,
126                    );
127                }
128                continue;
129            };
130
131            if dy < dy2 && dy >= dy1 {
132                let intersect = (dy - dy1) * (dx2 - dx1) / (dy2 - dy1) + dx1;
133                xs.push((intersect + 0.5).floor() as isize);
134            }
135        }
136
137        xs.sort_unstable();
138        for pair in xs[..].chunks_exact(2) {
139            let x_start = pair[0].max(min_x);
140            let x_end = pair[1].min((max_x + 1) as isize);
141            if x_start <= (max_x as isize) && x_end > min_x {
142                rasterizer.fill_horizontal_line(x_start as usize, x_end as usize, y as usize);
143            }
144        }
145        xs.clear();
146    }
147
148    // All the code above this point only deals with the interior of
149    // the polygon but this bit handles the exterior border.
150    once(first)
151        .chain(rest)
152        .for_each(|ls| ls.rasterize(rasterizer));
153}
